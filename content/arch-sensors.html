Sensor fusion

# Sensor fusion: correlating RF + Radar + Thermal detections

class SensorFusionEngine:

    CORRELATION_WINDOW_MS = 5000   # 5 second window
    CONFIDENCE_THRESHOLD  = 0.85

    def fuse_detections(self, detections: List[SensorEvent]) -> ThreatObject:
        clusters = self.spatiotemporal_cluster(detections)

        for cluster in clusters:
            confidence = self.compute_fusion_confidence(cluster)

            if confidence >= self.CONFIDENCE_THRESHOLD:
                return ThreatObject(
                    lat=cluster.centroid_lat,
                    lon=cluster.centroid_lon,
                    altitude=cluster.estimated_alt,
                    threat_class=self.classify_threat(cluster),
                    confidence=confidence,
                    velocity_ms=cluster.velocity,
                    heading=cluster.heading,
                    estimated_impact=self.project_trajectory(cluster)
                )
