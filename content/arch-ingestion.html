Ingestion pipeline

# Example: Satellite ingestion pipeline (Python)

class SatelliteIngestionPipeline:

    def __init__(self):
        self.iceye_client   = ICEYEClient(api_key=env.ICEYE_KEY)
        self.planet_client  = PlanetClient(api_key=env.PLANET_KEY)
        self.kafka_producer = KafkaProducer(bootstrap_servers=KAFKA_BROKERS)

    async def ingest_sar_imagery(self, aoi: GeoJSON) -> None:
        # Area of interest: target regions for the local government
        scenes = await self.iceye_client.search(
            aoi=aoi,
            start_time=datetime.utcnow() - timedelta(hours=2),
            product_type="SLC"
        )
        for scene in scenes:
            imagery  = await self.iceye_client.download(scene.id)
            metadata = self.extract_metadata(imagery)
            # Publish to Kafka for downstream AI processing
            self.kafka_producer.send(
                topic="satellite.sar.raw",
                value=ImageryPayload(scene=imagery, meta=metadata)
            )
